86class lazyattr(object): 

87 """Lazy object attribute whose value is computed on first access.""" 

88 __slots__ = ['func'] 

89 

90 def __init__(self, func): 

91 self.func = func 

92 

93 def __get__(self, instance, owner): 

94 result = self.func(instance) 

95 if result is NotImplemented: 

96 return getattr(super(owner, instance), self.func.__name__) 

97 setattr(instance, self.func.__name__, result) 

98 return result

101class Element(object): 

102 """Chemical element. 

103 

104 Attributes 

105 ---------- 

106 number : int 

107 Atomic number 

108 symbol : str of length 1 or 2 

109 Chemical symbol 

110 name : str 

111 Name in english

113 Group in periodic table

115 Period in periodic table 

116 block : int 

117 Block in periodic table 

118 series : int 

119 Index to chemical series 

120 protons : int 

121 Number of protons 

122 neutrons : int 

123 Number of neutrons in the most abundant naturally occurring stable 

124 isotope 

125 nominalmass : int 

126 Mass number of the most abundant naturally occurring stable isotope 

127 electrons : int 

128 Number of electrons 

129 mass : float 

130 Relative atomic mass. Ratio of the average mass of atoms 

131 of the element to 1/12 of the mass of an atom of 12C 

132 exactmass : float 

133 Relative atomic mass calculated from the isotopic composition 

134 eleneg : float 

135 Electronegativity (Pauling scale) 

136 covrad : float 

137 Covalent radius in Angstrom 

138 atmrad : 

139 Atomic radius in Angstrom 

140 vdwrad : float 

141 Van der Waals radius in Angstrom 

142 tboil : float 

143 Boiling temperature in K 

144 tmelt : float 

145 Melting temperature in K 

146 density : float 

147 Density at 295K in g/cm3 respectively g/L 

148 oxistates : str 

149 Oxidation states 

150 eleaffin : float 

151 Electron affinity in eV 

152 eleconfig : str 

153 Ground state electron configuration 

154 eleconfig_dict : dict 

155 Ground state electron configuration (shell, subshell): electrons 

156 eleshells : int 

157 Number of electrons per shell 

158 ionenergy : tuple 

159 Ionization energies in eV 

160 isotopes : dict 

161 Isotopic composition. 

162 keys: isotope mass number 

163 values: Isotope(relative atomic mass, abundance)

168 self.symbol = symbol 

169 self.name = name 

170 self.electrons = number

178 ionenergy = []

216 @lazyattr 

217 def nominalmass(self): 

218 """Return mass number of most abundant natural stable isotope.""" 

219 nominalmass = 0 

220 maxabundance = 0 

221 for massnum, iso in self.isotopes.items(): 

222 if iso.abundance > maxabundance: 

223 maxabundance = iso.abundance 

224 nominalmass = massnum 

225 return nominalmass

227 @lazyattr 

228 def neutrons(self): 

229 """Return number neutrons in most abundant natural stable isotope.""" 

230 return self.nominalmass - self.protons 

231 

232 @lazyattr 

233 def exactmass(self): 

234 """Return relative atomic mass calculated from isotopic composition.""" 

235 return sum(iso.mass * iso.abundance for iso in self.isotopes.values()) 

236 

237 @lazyattr 

238 def eleconfig_dict(self): 

239 """Return electron configuration as dict.""" 

240 adict = {} 

241 if self.eleconfig.startswith('['): 

242 base = self.eleconfig.split(' ', 1)[0][1:-1] 

243 adict.update(ELEMENTS[base].eleconfig_dict) 

244 for e in self.eleconfig.split()[bool(adict):]: 

245 adict[(int(e[0]), e[1])] = int(e[2:]) if len(e) > 2 else 1 

246 return adict

248 @lazyattr 

249 def eleshells(self): 

250 """Return number of electrons in shell as tuple.""" 

251 eleshells = [0, 0, 0, 0, 0, 0, 0] 

252 for key, val in self.eleconfig_dict.items(): 

253 eleshells[key[0] - 1] += val 

254 return tuple(ele for ele in eleshells if ele)

256 @lazyattr 

257 def description(self): 

258 """Return text description of element.""" 

259 return _descriptions(self.symbol)

261 def validate(self): 

262 """Check consistency of data. Raise Error on failure.""" 

263 assert self.period in PERIODS 

264 assert self.group in GROUPS 

265 assert self.block in BLOCKS 

266 assert self.series in SERIES 

267 

268 if self.number != self.protons: 

269 raise ValueError( 

270 '%s - atomic number must equal proton number' % self.symbol) 

271 if self.protons != sum(self.eleshells): 

272 raise ValueError( 

273 '%s - number of protons must equal electrons' % self.symbol) 

274 

275 if len(self.ionenergy) > 1: 

276 ionev_ = self.ionenergy[0] 

277 for ionev in self.ionenergy[1:]: 

278 if ionev <= ionev_: 

279 raise ValueError( 

280 '%s - ionenergy not increasing' % self.symbol) 

281 ionev_ = ionev 

282 

283 mass = 0.0 

284 frac = 0.0 

285 for iso in self.isotopes.values(): 

286 mass += iso.abundance * iso.mass 

287 frac += iso.abundance 

288 if abs(mass - self.mass) > 0.03: 

289 raise ValueError( 

290 '%s - average of isotope masses (%.4f) != mass (%.4f)' % ( 

291 self.symbol, mass, self.mass)) 

292 if abs(frac - 1.0) > 1e-9: 

293 raise ValueError( 

294 '%s - sum of isotope abundances != 1.0' % self.symbol)

315class ElementsDict(object): 

316 """Ordered dict of Elements with lookup by number, symbol, and name.""" 

317 def __init__(self, *elements): 

318 self._list = [] 

319 self._dict = {} 

320 for element in elements: 

321 if element.number > len(self._list) + 1: 

322 raise ValueError('Elements must be added in order') 

323 if element.number <= len(self._list): 

324 self._list[element.number - 1] = element 

325 else: 

326 self._list.append(element) 

327 self._dict[element.number] = element 

328 self._dict[element.symbol] = element 

329 self._dict[element.name] = element 

330 

331 def __str__(self): 

332 return '[%s]' % ', '.join(ele.symbol for ele in self._list) 

333 

334 def __contains__(self, item): 

335 return item in self._dict 

336 

337 def __iter__(self): 

338 return iter(self._list) 

339 

340 def __len__(self): 

341 return len(self._list) 

342 

343 def __getitem__(self, key): 

344 try: 

345 return self._dict[key] 

346 except KeyError: 

347 try: 

348 start, stop, step = key.indices(len(self._list)) 

349 return self._list[slice(start - 1, stop - 1, step)] 

350 except Exception: 

351 raise KeyError
