244 def get_accuracy(self, prediction, target, reduce=True): 

245 ''' 

246 Computes the accuracy for classification tasks 

247 

248 prediction : tensor of torch.Size([batch_size, number of classes]) 

249 The prediction tensor contains softmax activation function output 

250 i.e. probabilities of all classes 

251 Ex : tensor([[8.5442e-30, 1.0000e+00, 4.4564e-27, 1.4013e-45, 0.0000e+00], 

252 [8.7185e-10, 3.6507e-08, 1.0244e-05, 5.2405e-10, 9.9999e-01], 

253 [4.7920e-29, 1.0000e+00, 2.4772e-27, 0.0000e+00, 0.0000e+00], 

254 [0.0000e+00, 1.0000e+00, 0.0000e+00, 0.0000e+00, 0.0000e+00]]) 

255 

256 target : tensor of torch.Size([batch_size]) 

257 Ex : tensor([1, 4, 0, 1]) 

258 

259 prediction.argmax(dim=1) returns the indices of the maximum values along the dim 1 

260 e.i. the class with the highest probability 

261 Ex : tensor([1, 4, 1, 1]) 

262 

263 (prediction.argmax(dim=1)==target) 

264 compares the content of the two tensors and returns a True/False tensor 

265 Ex : tensor([True, True, False, True]) 

266 

267 overlap = (prediction.argmax(dim=1)==target).sum() 

268 counts the number of True booleans 

269 

270 overlap/float(target.size()[-1]) 

271 divides the number of True booleans by the batch size 

272 and thus returns an accuracy value 

273 Ex : tensor(0.7500) 

274 ''' 

275 overlap = (prediction.argmax(dim=1) == target).sum() 

276 if reduce: 

277 return overlap/float(target.size()[-1]) 

278 

279 return overlap
