60 

61 Returns:

335 def compute_Fnat_fast(self, ref_pairs=None, cutoff=5): 

336 """Compute the FNAT of the conformation

349 if ref_pairs is None: 

350 residue_pairs_ref = self.compute_residue_pairs_ref(

367 data_decoy = f.readlines() 

368 decoy_name = os.path.basename(self.decoy) 

369 elif isinstance(self.decoy, np.ndarray): 

370 data_decoy = [l.decode('utf-8') for l in self.decoy] 

371 decoy_name = 'decoy' 

372 else: 

373 raise ValueError('Decoy not found')

382 for line in data_decoy: 

383 

384 if line.startswith('ATOM'): 

385 

386 # chain ID 

387 chainID = line[21] 

388 if chainID == ' ': 

389 chainID = line[72] 

390

392 resSeq = int(line[22:26]) 

393 resName = line[17:20].strip() 

394 name = line[12:16].strip() 

395

398 line[38:46]), float(line[46:54])

402

466 

467 This routine parse the PDB directly using pdb2sql. 

468 L-RMSD is computed by aligning the longest chain of the decoy to the one of the reference 

469 and computing the RMSD of the shortest chain between decoy and reference.

472 http://journals.plos.org/plosone/article?id=10.1371/journal.pone.0161879 

473 

474 Args: 

475 exportpath (str, optional): file name where the aligned pdbs are exported 

476 method (str, optional): Method to align the fragments ('svd','quaternion') 

477 

478 Returns: 

479 float: L-RMSD value of the conformation 

480 """ 

481

483 sql_decoy = pdb2sql(self.decoy, sqlfile='decoy.db') 

484 sql_ref = pdb2sql(self.ref, sqlfile='ref.db') 

485 

486 # extract the pos of chains A 

487 xyz_decoy_A = np.array(sql_decoy.get('x,y,z', chainID='A')) 

488 xyz_ref_A = np.array(sql_ref.get('x,y,z', chainID='A')) 

489 

490 # extract the pos of chains B 

491 xyz_decoy_B = np.array(sql_decoy.get('x,y,z', chainID='B')) 

492 xyz_ref_B = np.array(sql_ref.get('x,y,z', chainID='B')) 

493 

494 # check the lengthes 

495 if len(xyz_decoy_A) != len(xyz_ref_A): 

496 xyz_decoy_A, xyz_ref_A = self.get_identical_atoms( 

497 sql_decoy, sql_ref, 'A') 

498 

499 if len(xyz_decoy_B) != len(xyz_ref_B): 

500 xyz_decoy_B, xyz_ref_B = self.get_identical_atoms( 

501 sql_decoy, sql_ref, 'B') 

502 

503 # detect which chain is the longest 

504 nA, nB = len(xyz_decoy_A), len(xyz_decoy_B) 

505 if nA > nB: 

506 xyz_decoy_long = xyz_decoy_A 

507 xyz_ref_long = xyz_ref_A 

508 

509 xyz_decoy_short = xyz_decoy_B 

510 xyz_ref_short = xyz_ref_B 

511 

512 else: 

513 xyz_decoy_long = xyz_decoy_B 

514 xyz_ref_long = xyz_ref_B 

515 

516 xyz_decoy_short = xyz_decoy_A 

517 xyz_ref_short = xyz_ref_A 

518

520 tr_decoy = self.get_trans_vect(xyz_decoy_long) 

521 tr_ref = self.get_trans_vect(xyz_ref_long) 

522

524 xyz_decoy_short = self.translation(xyz_decoy_short, tr_decoy) 

525 xyz_decoy_long = self.translation(xyz_decoy_long, tr_decoy) 

526 

527 # translate everuthing for 2 

528 xyz_ref_short = self.translation(xyz_ref_short, tr_ref) 

529 xyz_ref_long = self.translation(xyz_ref_long, tr_ref) 

530 

531 # get the ideal rotation matrix 

532 # to superimpose the A chains 

533 U = self.get_rotation_matrix( 

534 xyz_decoy_long, xyz_ref_long, method=method) 

535 

536 # rotate the entire fragment 

537 xyz_decoy_short = self.rotation_matrix( 

538 xyz_decoy_short, U, center=False) 

539 

540 # compute the RMSD 

541 lrmsd = self.get_rmsd(xyz_decoy_short, xyz_ref_short) 

542 

543 # export the pdb for verifiactions 

544 if exportpath is not None: 

545 

546 # extract the pos of the dimer 

547 xyz_decoy = np.array(sql_decoy.get('x,y,z')) 

548 xyz_ref = np.array(sql_ref.get('x,y,z')) 

549 

550 # translate 

551 xyz_ref = self.translation(xyz_ref, tr_ref) 

552 xyz_decoy = self.translation(xyz_decoy, tr_decoy) 

553 

554 # rotate decoy 

555 xyz_decoy = self.rotation_matrix( 

556 xyz_decoy, U, center=False) 

557 

558 # update the sql database 

559 sql_decoy.update_xyz(xyz_decoy) 

560 sql_ref.update_xyz(xyz_ref) 

561

564 sql_ref.exportpdb(exportpath+'/lrmsd_aligned.pdb')

608 if izone is not given in argument the routine will compute them automatically 

609 

610 i-RMSD is computed selecting the back bone of the contact residue with a cutoff of 10A 

611 in the decoy. Align these as best as possible with their coutner part in the ref 

612 and and compute the RMSD 

613 

614 Ref : DockQ: A Quality Measure for Protein-Protein Docking Models 

615 http://journals.plos.org/plosone/article?id=10.1371/journal.pone.0161879 

616 

617 Args: 

618 izone (None, optional): file name of the zone. if None the zones will be calculated first 

619 method (str, optional): Method to align the fragments ('svd','quaternion') 

620 cutoff (float, optional): cutoff for the contact atoms 

621 exportpath (str, optional): file name where the aligned pdbs are exported 

622 

623 Returns: 

624 float: i-RMSD value of the conformation 

625 """ 

626 

627 # create thes sql 

628 sql_decoy = pdb2sql(self.decoy) 

629 sql_ref = pdb2sql(self.ref) 

630 

631 # get the contact atoms 

632 if izone is None: 

633 contact_ref = sql_ref.get_contact_atoms(cutoff=cutoff, extend_to_residue=True, 

634 return_only_backbone_atoms=False) 

635 index_contact_ref = contact_ref[0]+contact_ref[1] 

636 else: 

637 index_contact_ref = self.get_izone_rowID( 

638 sql_ref, izone, return_only_backbone_atoms=False) 

639 

640 # get the xyz and atom identifier of the decoy contact atoms 

641 xyz_contact_ref = sql_ref.get( 

642 'x,y,z', rowID=index_contact_ref) 

643 data_contact_ref = sql_ref.get( 

644 'chainID,resSeq,resName,name', rowID=index_contact_ref) 

645 

646 # get the xyz and atom indeitifier of the reference 

647 xyz_decoy = sql_decoy.get('x,y,z') 

648 data_decoy = sql_decoy.get('chainID,resSeq,resName,name') 

649 

650 # loop through the ref label 

651 # check if the atom is in the decoy 

652 # if yes -> add xyz to xyz_contact_decoy 

653 # if no -> remove the corresponding to xyz_contact_ref 

654 xyz_contact_decoy = [] 

655 index_contact_decoy = [] 

656 clean_ref = False 

657 for iat, atom in enumerate(data_contact_ref): 

658 

659 try: 

660 index = data_decoy.index(atom) 

661 index_contact_decoy.append(index) 

662 xyz_contact_decoy.append(xyz_decoy[index]) 

663 except Exception: 

664 xyz_contact_ref[iat] = None 

665 index_contact_ref[iat] = None 

666 clean_ref = True 

667 

668 # clean the xyz 

669 if clean_ref: 

670 xyz_contact_ref = [ 

671 xyz for xyz in xyz_contact_ref if xyz is not None] 

672 index_contact_ref = [ 

673 ind for ind in index_contact_ref if ind is not None] 

674 

675 # check that we still have atoms in both chains 

676 chain_decoy = list( 

677 set(sql_decoy.get('chainID', rowID=index_contact_decoy))) 

678 chain_ref = list( 

679 set(sql_ref.get('chainID', rowID=index_contact_ref))) 

680 

681 if len(chain_decoy) < 1 or len(chain_ref) < 1: 

682 raise ValueError( 

683 'Error in i-rmsd: only one chain represented in one chain') 

684 

685 # get the translation so that both A chains are centered 

686 tr_decoy = self.get_trans_vect(xyz_contact_decoy) 

687 tr_ref = self.get_trans_vect(xyz_contact_ref) 

688 

689 # translate everything 

690 xyz_contact_decoy = self.translation( 

691 xyz_contact_decoy, tr_decoy) 

692 xyz_contact_ref = self.translation(xyz_contact_ref, tr_ref) 

693 

694 # get the ideql rotation matrix 

695 # to superimpose the A chains 

696 U = self.get_rotation_matrix( 

697 xyz_contact_decoy, xyz_contact_ref, method=method) 

698 

699 # rotate the entire fragment 

700 xyz_contact_decoy = self.rotation_matrix( 

701 xyz_contact_decoy, U, center=False) 

702 

703 # compute the RMSD 

704 irmsd = self.get_rmsd(xyz_contact_decoy, xyz_contact_ref) 

705 

706 

707 # export the pdb for verifiactions 

708 if exportpath is not None: 

709 

710 # update the sql database 

711 sql_decoy.update_xyz( 

712 xyz_contact_decoy, index=index_contact_decoy) 

713 sql_ref.update_xyz( 

714 xyz_contact_ref, index=index_contact_ref) 

715 

716 sql_decoy.exportpdb( 

717 exportpath+'/irmsd_decoy.pdb', index=index_contact_decoy) 

718 sql_ref.exportpdb( 

719 exportpath+'/irmsd_ref.pdb', index=index_contact_ref) 

720 

721 # close the db 

722 sql_decoy.close() 

723 sql_ref.close() 

724 

725 return irmsd

1051 @staticmethod 

1052 def get_rotation_matrix_Kabsh(P, Q): 

1053 '''Get the rotation matrix to aligh two point clouds. 

1054 

1055 The method is based on th Kabsh approach 

1056 https://cnx.org/contents/HV-RsdwL@23/Molecular-Distance-Measures 

1057 

1058 Args: 

1059 P (np.array): xyz of the first point cloud 

1060 Q (np.array): xyz of the second point cloud 

1061 

1062 Returns: 

1063 np.array: rotation matrix 

1064 

1065 Raises: 

1066 ValueError: matrix have different sizes 

1067 ''' 

1068 

1069 pshape = P.shape 

1070 qshape = Q.shape 

1071 

1072 if pshape[0] == qshape[0]: 

1073 npts = pshape[0] 

1074 else: 

1075 raise ValueError( 

1076 "Matrix don't have the same number of points", P.shape, Q.shape) 

1077 

1078 p0, q0 = np.abs(np.mean(P, 0)), np.abs(np.mean(Q, 0)) 

1079 eps = 1E-6 

1080 if any(p0 > eps) or any(q0 > eps): 

1081 raise ValueError( 

1082 'You must center the fragment first', p0, q0) 

1083 

1084 # form the covariance matrix 

1085 A = np.dot(P.T, Q)/npts 

1086 

1087 # SVD the matrix 

1088 V, S, W = np.linalg.svd(A) 

1089 

1090 # the W matrix returned here is 

1091 # already its transpose 

1092 # https://docs.scipy.org/doc/numpy-1.13.0/reference/generated/numpy.linalg.svd.html 

1093 W = W.T 

1094 

1095 # determinant 

1096 d = np.linalg.det(np.dot(W, V.T)) 

1097 

1098 # form the U matrix 

1099 Id = np.eye(3) 

1100 if d < 0: 

1101 Id[2, 2] = -1 

1102 

1103 U = np.dot(W, np.dot(Id, V.T)) 

1104 

1105 return U 

1106 

1107 @staticmethod 

1108 def get_rotation_matrix_quaternion(P, Q): 

1109 '''Get the rotation matrix to aligh two point clouds 

1110 

1111 The method is based on the quaternion approach 

1112 http://www.ams.stonybrook.edu/~coutsias/papers/rmsd17.pdf 

1113 

1114 Args: 

1115 P (np.array): xyz of the first point cloud 

1116 Q (np.array): xyz of the second point cloud 

1117 

1118 Returns: 

1119 np.array: rotation matrix 

1120 

1121 Raises: 

1122 ValueError: matrix have different sizes 

1123 ''' 

1124 

1125 pshape = P.shape 

1126 qshape = Q.shape 

1127 

1128 if pshape[0] != qshape[0]: 

1129 raise ValueError( 

1130 "Matrix don't have the same number of points", P.shape, Q.shape) 

1131 

1132 p0, q0 = np.abs(np.mean(P, 0)), np.abs(np.mean(Q, 0)) 

1133 eps = 1E-6 

1134 if any(p0 > eps) or any(q0 > eps): 

1135 raise ValueError( 

1136 'You must center the fragment first', p0, q0) 

1137 

1138 # form the correlation matrix 

1139 R = np.dot(P.T, Q) 

1140 

1141 # form the F matrix (eq. 10 of ref[1]) 

1142 F = np.zeros((4, 4)) 

1143 

1144 F[0, 0] = np.trace(R) 

1145 F[0, 1] = R[1, 2]-R[2, 1] 

1146 F[0, 2] = R[2, 0]-R[0, 2] 

1147 F[0, 3] = R[0, 1]-R[1, 0] 

1148 

1149 F[1, 0] = R[1, 2]-R[2, 1] 

1150 F[1, 1] = R[0, 0]-R[1, 1]-R[2, 2] 

1151 F[1, 2] = R[0, 1]+R[1, 0] 

1152 F[1, 3] = R[0, 2]+R[2, 0] 

1153 

1154 F[2, 0] = R[2, 0]-R[0, 2] 

1155 F[2, 1] = R[0, 1]+R[1, 0] 

1156 F[2, 2] = -R[0, 0]+R[1, 1]-R[2, 2] 

1157 F[2, 3] = R[1, 2]+R[2, 1] 

1158 

1159 F[3, 0] = R[0, 1]-R[1, 0] 

1160 F[3, 1] = R[0, 2]+R[2, 0] 

1161 F[3, 2] = R[1, 2]+R[2, 1] 

1162 F[3, 3] = -R[0, 0]-R[1, 1]+R[2, 2] 

1163 

1164 # diagonalize it 

1165 l, U = np.linalg.eig(F) 

1166 

1167 # extract the eigenvect of the highest eigenvalues 

1168 indmax = np.argmax(l) 

1169 q0, q1, q2, q3 = U[:, indmax] 

1170 

1171 # form the rotation matrix (eq. 33 ref[1]) 

1172 U = np.zeros((3, 3)) 

1173 

1174 U[0, 0] = q0**2+q1**2-q2**2-q3**2 

1175 U[0, 1] = 2*(q1*q2-q0*q3) 

1176 U[0, 2] = 2*(q1*q3+q0*q2) 

1177 U[1, 1] = 2*(q1*q2+q0*q3) 

1178 U[1, 2] = q0**2-q1**2+q2*2-q3**2 

1179 U[1, 2] = 2*(q2*q3-q0*q1) 

1180 U[2, 0] = 2*(q1*q3-q0*q2) 

1181 U[2, 1] = 2*(q2*q3+q0*q1) 

1182 U[2, 2] = q0**2-q1**2-q2**2+q3**2 

1183 

1184 return U

1199 @staticmethod 

1200 def rotation_around_axis(xyz, axis, angle): 

1201 """Rotate a fragment around an axis. 

1202 

1203 Args: 

1204 xyz (np.array): original positions 

1205 axis (np.array): axis of rotation 

1206 angle (float): angle of rotation (radians) 

1207 

1208 Returns: 

1209 np.array: Rotated positions 

1210 """ 

1211 

1212 # get the data 

1213 ct, st = np.cos(angle), np.sin(angle) 

1214 ux, uy, uz = axis 

1215 

1216 # get the center of the molecule 

1217 xyz0 = np.mean(xyz, 0) 

1218 

1219 # definition of the rotation matrix 

1220 # see https://en.wikipedia.org/wiki/Rotation_matrix 

1221 rot_mat = np.array([ 

1222 [ct + ux**2*(1-ct), ux*uy*(1-ct) - 

1223 uz*st, ux*uz*(1-ct) + uy*st], 

1224 [uy*ux*(1-ct) + uz*st, ct + uy**2 * 

1225 (1-ct), uy*uz*(1-ct) - ux*st], 

1226 [uz*ux*(1-ct) - uy*st, uz*uy*(1-ct) + ux*st, ct + uz**2*(1-ct)]]) 

1227 

1228 # apply the rotation 

1229 return np.dot(rot_mat, (xyz-xyz0).T).T + xyz0

1244 # precompute the trig 

1245 ca, sa = np.cos(alpha), np.sin(alpha) 

1246 cb, sb = np.cos(beta), np.sin(beta) 

1247 cg, sg = np.cos(gamma), np.sin(gamma) 

1248 

1249 # get the center of the molecule 

1250 xyz0 = np.mean(xyz, 0) 

1251 

1252 # rotation matrices 

1253 rx = np.array([[1, 0, 0], [0, ca, -sa], [0, sa, ca]]) 

1254 ry = np.array([[cb, 0, sb], [0, 1, 0], [-sb, 0, cb]]) 

1255 rz = np.array([[cg, -sg, 0], [sg, cs, 0], [0, 0, 1]]) 

1256 

1257 rot_mat = np.dot(rx, np.dot(ry, rz)) 

1258 

1259 # apply the rotation 

1260 return np.dot(rot_mat, (xyz-xyz0).T).T + xyz0

1262 @staticmethod 

1263 def rotation_matrix(xyz, rot_mat, center=True): 

1264 """Rotate a fragment from a roation matrix 

1265 

1266 Args: 

1267 xyz (np.array): original positions 

1268 rot_mat (np.array): rotation matrix 

1269 center (bool, optional): Center the fragment before rotation 

1270 

1271 Returns: 

1272 np.array: rotated positions 

1273 """ 

1274 if center: 

1275 xyz0 = np.mean(xyz) 

1276 return np.dot(rot_mat, (xyz-xyz0).T).T + xyz0 

1277 else: 

1278 return np.dot(rot_mat, (xyz).T).T
